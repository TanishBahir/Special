<script>
// ----------------------
// STAR GENERATOR
// ----------------------

const stars = document.querySelector(".stars");

for(let i = 0; i < 180; i++){

    const star = document.createElement("div");
    star.className = "star";

    star.style.left = Math.random()*100 + "%";
    star.style.top = Math.random()*100 + "%";

    star.style.animationDelay = Math.random()*3 + "s";
    star.style.opacity = Math.random();

    stars.appendChild(star);

}


// ----------------------
// TYPEWRITER MESSAGE
// ----------------------

const message = `Dear Rhucha,

I don't know whether you'll read this till the end,
but thank you for opening it.

This isn't a request.
This isn't an expectation.

It's simply a few words
I wanted to leave behind.

Every conversation,
every smile,
every memory—
they became little stars
in my life.

Thank you for being a part
of my journey.

No matter where life takes us,
I genuinely hope
you stay happy,
healthy,
and keep smiling.

Some people become chapters.

Some become lessons.

And some become memories
that quietly stay in the heart forever.

Thank you...

— Tanish ❤️`;

const target = document.getElementById("typewriter");

let index = 0;

function typeWriter(){

    if(index < message.length){

        target.innerHTML += message.charAt(index);

        index++;

        setTimeout(typeWriter,35);

    }

}

typeWriter();

    const heart=document.createElement("div");

    heart.className="heart";

    heart.innerHTML="❤️";

    heart.style.left=e.clientX+"px";
    heart.style.top=e.clientY+"px";

    heart.style.fontSize=(18+Math.random()*20)+"px";

    document.body.appendChild(heart);

    setTimeout(()=>{
        heart.remove();
    },4000);

});

// ----------------------
// SHOOTING STARS
// ----------------------

function createShootingStar(){

    const star = document.createElement("div");

    star.style.position = "absolute";
    star.style.width = "120px";
    star.style.height = "2px";
    star.style.background =
        "linear-gradient(90deg, white, transparent)";
    star.style.left = Math.random()*80 + "%";
    star.style.top = Math.random()*40 + "%";
    star.style.transform = "rotate(-35deg)";
    star.style.opacity = "0.9";
    star.style.pointerEvents = "none";
    star.style.zIndex = "1";

    document.body.appendChild(star);

    let x = 0;

    const interval = setInterval(()=>{

        x += 18;

        star.style.transform =
            `translate(${x}px,${x}px) rotate(-35deg)`;

        star.style.opacity -= 0.02;

        if(x > 700){

            clearInterval(interval);
            star.remove();

        }

    },16);

}

setInterval(createShootingStar,5000);

    glow.style.top=e.clientY+"px";
    glow.style.width="8px";

    glow.style.borderRadius="50%";
    glow.style.background="white";
    glow.style.boxShadow="0 0 20px white";
    glow.style.pointerEvents="none";
    glow.style.opacity=".8";

    document.body.appendChild(glow);

    setTimeout(()=>{
        glow.style.transition=".7s";
        glow.style.opacity="0";
        glow.style.transform="scale(4)";
    },10);

    setTimeout(()=>{
        glow.remove();
    },700);

});
